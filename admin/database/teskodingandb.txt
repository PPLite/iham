SELECT 
tb_reader_scan.rfid_uid,

tb_rfid.nama_alat,
tb_rfid.deskripsi,
tb_rfid.penanggung_jawab,
tb_rfid.peminjam,
tb_rfid.status_asset,
tb_rfid.tanggal,

tb_stat_anak.id_pengenal,
tb_stat_anak.nama_anak,
tb_stat_anak.nama_ibu,
tb_stat_anak.penanggung_jawab,
tb_stat_anak.alamat,
tb_stat_anak.status,
tb_stat_anak.waktu_masuk

FROM tb_reader_scan 
LEFT JOIN tb_rfid ON tb_reader_scan.rfid_uid = tb_rfid.rfid_uid 
LEFT JOIN tb_stat_anak ON tb_reader_scan.rfid_uid = tb_stat_anak.rfid_uid 
ORDER BY tb_reader_scan.rfid_uid
